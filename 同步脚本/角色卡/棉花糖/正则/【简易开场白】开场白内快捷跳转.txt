/* ═══════════════════════════════════════════

(function(){
  var container=document.getElementById('jk-wrap');
  var canvas=document.getElementById('jk-canvas');
  if(!container||!canvas)return;

  var ctx=canvas.getContext('2d'), w=0, h=0;
  var ambient=[], cursorParts=[], trail=[];
  var AMBIENT=35, MAX_CURSOR=50, MAX_TRAIL=22;
  var REPEL_R=48, REPEL_FORCE=.40;
  var mouseX=-200,mouseY=-200,mouseOn=false,mouseSpeed=0,lastMX=-200,lastMY=-200;

  function rnd(a,b){return a+Math.random()*(b-a)}

  function resize(){
    var r=container.getBoundingClientRect();
    w=r.width; h=r.height;
    canvas.width=w; canvas.height=h;
    initAmbient();
  }

  function initAmbient(){
    ambient=[];
    for(var i=0;i<AMBIENT;i++){
      ambient.push({
        x:Math.random()*w, y:Math.random()*h,
        vx:(Math.random()-.5)*.15,
        vy:.08+Math.random()*.35,
        baseVy:.08+Math.random()*.35,
        size:.4+Math.random()*1.4,
        pink:Math.random()<.62,
        phase:Math.random()*Math.PI*2,
        sparkTimer:Math.random()*8
      });
    }
  }

  function spawnCursor(x,y,speed){
    var count=speed>2.5?4:(speed>.8?3:2);
    for(var i=0;i<count;i++){
      if(cursorParts.length>=MAX_CURSOR) cursorParts.shift();
      cursorParts.push({
        x:x+rnd(-3,3), y:y+rnd(-3,3),
        vx:rnd(-.6,.6), vy:rnd(-.6,.6),
        size:.5+rnd(.3,2.2), life:1,
        decay:.012+rnd(.006,.032),
        pink:Math.random()<.58
      });
    }
  }

  var wisps=[];
  for(var wi=0;wi<3;wi++){
    wisps.push({
      x1:rnd(.05,.35)*w, y1:rnd(.1,.9)*h,
      cx1:rnd(.1,.5)*w, cy1:rnd(0,1)*h,
      cx2:rnd(.5,.9)*w, cy2:rnd(0,1)*h,
      x2:rnd(.65,.95)*w, y2:rnd(.1,.9)*h,
      vx1:rnd(-.15,.15), vy1:rnd(-.1,.1),
      vx2:rnd(-.15,.15), vy2:rnd(-.1,.1),
      vx3:rnd(-.12,.12), vy3:rnd(-.08,.08),
      vx4:rnd(-.12,.12), vy4:rnd(-.08,.08),
      pink:wi<2, opacity:rnd(.025,.055)
    });
  }

  function update(){
    if(mouseOn){
      trail.push({x:mouseX,y:mouseY});
      if(trail.length>MAX_TRAIL) trail.shift();
      var cnt=mouseSpeed>3?4:(mouseSpeed>1?3:2);
      for(var i=0;i<cnt;i++) spawnCursor(mouseX,mouseY,mouseSpeed);
      if(trail.length>2){
        var step=Math.max(2,Math.floor(trail.length/6));
        for(var k=0;k<trail.length-step;k+=step){
          if(Math.random()<.25) spawnCursor(trail[k].x,trail[k].y,1);
        }
      }
    }else{if(trail.length>0) trail.shift()}

    for(var j=cursorParts.length-1;j>=0;j--){
      var cp=cursorParts[j];
      cp.x+=cp.vx; cp.y+=cp.vy;
      cp.vx*=.96; cp.vy*=.96;
      cp.life-=cp.decay;
      if(cp.life<=0) cursorParts.splice(j,1);
    }

    var samples=[];
    if(mouseOn) samples.push({x:mouseX,y:mouseY});
    for(var s=0;s<trail.length;s+=4) samples.push(trail[s]);

    for(var m=0;m<ambient.length;m++){
      var ap=ambient[m];
      if(samples.length>0&&mouseOn){
        var tfx=0,tfy=0;
        for(var n=0;n<samples.length;n++){
          var dx=ap.x-samples[n].x, dy=ap.y-samples[n].y;
          var dist=Math.sqrt(dx*dx+dy*dy);
          if(dist<REPEL_R&&dist>.3){
            var f=REPEL_FORCE*Math.pow(1-dist/REPEL_R,1.5);
            tfx+=(dx/dist)*f; tfy+=(dy/dist)*f;
          }
        }
        ap.vx+=tfx; ap.vy+=tfy;
      }
      ap.x+=ap.vx+Math.sin(ap.phase+performance.now()*.001)*.08;
      ap.y+=ap.vy;
      ap.phase+=.007;
      ap.vx*=.984; ap.vy*=.984;
      ap.vy+=(ap.baseVy-ap.vy)*.003;
      ap.sparkTimer+=.016;
      if(ap.y>h+12){ap.y=-12; ap.x=Math.random()*w; ap.vx=0; ap.vy=ap.baseVy}
      if(ap.x<-12) ap.x=w+12; if(ap.x>w+12) ap.x=-12;
    }

    for(var wi2=0;wi2<wisps.length;wi2++){
      var ws=wisps[wi2];
      ws.cx1+=ws.vx1; ws.cy1+=ws.vy1;
      ws.cx2+=ws.vx2; ws.cy2+=ws.vy2;
      ws.x1+=ws.vx3; ws.y1+=ws.vy3;
      ws.x2+=ws.vx4; ws.y2+=ws.vy4;
      if(ws.cx1<0||ws.cx1>w) ws.vx1*=-1; if(ws.cy1<0||ws.cy1>h) ws.vy1*=-1;
      if(ws.cx2<0||ws.cx2>w) ws.vx2*=-1; if(ws.cy2<0||ws.cy2>h) ws.vy2*=-1;
      if(ws.x1<0||ws.x1>w) ws.vx3*=-1; if(ws.y1<0||ws.y1>h) ws.vy3*=-1;
      if(ws.x2<0||ws.x2>w) ws.vx4*=-1; if(ws.y2<0||ws.y2>h) ws.vy4*=-1;
    }

    if(mouseOn){
      mouseSpeed=Math.sqrt((mouseX-lastMX)*(mouseX-lastMX)+(mouseY-lastMY)*(mouseY-lastMY));
      lastMX=mouseX; lastMY=mouseY;
    }
  }

  function draw(){
    ctx.clearRect(0,0,w,h);

    for(var wi=0;wi<wisps.length;wi++){
      var ws=wisps[wi];
      var col=ws.pink?'240,168,200':'160,192,240';
      ctx.strokeStyle='rgba('+col+','+ws.opacity+')';
      ctx.lineWidth=.6;
      ctx.beginPath();
      ctx.moveTo(ws.x1,ws.y1);
      ctx.bezierCurveTo(ws.cx1,ws.cy1,ws.cx2,ws.cy2,ws.x2,ws.y2);
      ctx.stroke();
    }

    for(var i=0;i<ambient.length;i++){
      var ap=ambient[i];
      var spark=1+Math.sin(ap.sparkTimer)*.35;
      var a=(.18+Math.sin(ap.phase)*.08)*spark;
      var col=ap.pink?'236,168,198':'180,200,240';
      var g=ctx.createRadialGradient(ap.x,ap.y,0,ap.x,ap.y,ap.size*3);
      g.addColorStop(0,'rgba('+col+','+a+')');
      g.addColorStop(.4,'rgba('+col+','+(a*.25)+')');
      g.addColorStop(1,'rgba('+col+',0)');
      ctx.fillStyle=g;
      ctx.beginPath(); ctx.arc(ap.x,ap.y,ap.size*3,0,Math.PI*2); ctx.fill();
      ctx.fillStyle='rgba('+col+','+(a*1.5)+')';
      ctx.beginPath(); ctx.arc(ap.x,ap.y,ap.size*.35,0,Math.PI*2); ctx.fill();
    }

    ctx.globalCompositeOperation='lighter';
    for(var j=0;j<cursorParts.length;j++){
      var cp=cursorParts[j];
      var a2=cp.life*.6;
      var col2=cp.pink?'236,168,198':'180,200,240';
      var sz=cp.size*(1+cp.life*.6);
      var g2=ctx.createRadialGradient(cp.x,cp.y,0,cp.x,cp.y,sz*2.5);
      g2.addColorStop(0,'rgba('+col2+','+a2+')');
      g2.addColorStop(.3,'rgba('+col2+','+(a2*.5)+')');
      g2.addColorStop(1,'rgba('+col2+',0)');
      ctx.fillStyle=g2;
      ctx.beginPath(); ctx.arc(cp.x,cp.y,sz*2.5,0,Math.PI*2); ctx.fill();
      var cx2=a2*.35;
      ctx.fillStyle='rgba(255,255,255,'+cx2+')';
      ctx.fillRect(cp.x-sz*.08,cp.y-sz*.7,sz*.16,sz*1.4);
      ctx.fillRect(cp.x-sz*.7,cp.y-sz*.08,sz*1.4,sz*.16);
    }
    ctx.globalCompositeOperation='source-over';
  }

  function loop(){update();draw();requestAnimationFrame(loop)}

    clearTimeout(window._jkResize);

    mouseX=e.clientX-r.left; mouseY=e.clientY-r.top;
  });
  container.addEventListener('mouseleave',function(){mouseOn=false; trail=[];});
  loop();
})();
