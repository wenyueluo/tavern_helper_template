var petalCanvas=document.getElementById('wish-canvas');
var pctx=petalCanvas.getContext('2d');
var petals=[], petalRunning=false, petalSpawn=null;

function resizePetal(){
  var r=petalCanvas.parentNode.getBoundingClientRect();
  petalCanvas.width=r.width; petalCanvas.height=r.height;
}
resizePetal(); $(window).on('resize',resizePetal);

function rndP(a,b){return a+Math.random()*(b-a)}

function spawnPetal(x,y,burst){
  return{
    x:x||rndP(0,petalCanvas.width), y:y||rndP(-10,petalCanvas.height*.3),
    vx:rndP(-.5,.5)+(burst?rndP(-1.5,1.5):0),
    vy:rndP(.3,1.2)+(burst?rndP(.5,2):0),
    size:rndP(5,14)+(burst?rndP(2,8):0),
    rot:rndP(0,6.28), rotV:rndP(-.03,.03),
    life:1, decay:rndP(.002,.006),
    pink:Math.random()<.7,
    sway:rndP(.5,1.5), phase:rndP(0,6.28)
  };
}

function startPetalBurst(){
  if(petalRunning)return; petalRunning=true;
  petalCanvas.style.opacity='1'; petals=[]; resizePetal();
  var cx=petalCanvas.width*.5, cy=petalCanvas.height*.45;
  for(var i=0;i<50;i++) petals.push(spawnPetal(cx,cy,true));
  petalSpawn=setInterval(function(){
    if(!petalRunning){clearInterval(petalSpawn);return}
    if(petals.length>180)return;
    for(var k=0;k<4;k++) petals.push(spawnPetal());
  },80);
  if(!window._petalLoop) requestAnimationFrame(petalLoop);
}

function stopPetals(){petalRunning=false;if(petalSpawn){clearInterval(petalSpawn);petalSpawn=null}}

function petalLoop(){
  window._petalLoop=true;
  if(!petalRunning && petals.length===0){window._petalLoop=false;return}
  pctx.clearRect(0,0,petalCanvas.width,petalCanvas.height);
  for(var i=petals.length-1;i>=0;i--){
    var p=petals[i];
    p.x+=p.vx+Math.sin(p.phase+performance.now()*.001)*p.sway*.3;
    p.y+=p.vy; p.rot+=p.rotV; p.life-=p.decay;
    if(p.life<=0){petals.splice(i,1);continue}
    var a=p.life*.5;
    var hue=p.pink?342:215, sat=p.pink?'55%':'50%', light=p.pink?'78%':'72%';
    pctx.save();pctx.translate(p.x,p.y);pctx.rotate(p.rot);
    pctx.fillStyle='hsla('+hue+','+sat+','+light+','+a+')';
    pctx.beginPath();pctx.ellipse(0,0,p.size*.6,p.size*.28,0,0,6.2832);pctx.fill();
    var g=pctx.createRadialGradient(0,0,p.size*.15,0,0,p.size*.7);
    g.addColorStop(0,'hsla('+hue+','+sat+','+light+','+(a*.5)+')');
    g.addColorStop(1,'hsla('+hue+','+sat+','+light+',0)');
    pctx.fillStyle=g;pctx.beginPath();pctx.arc(0,0,p.size*.7,0,6.2832);pctx.fill();
    pctx.restore();
  }
  requestAnimationFrame(petalLoop);
}

(function(){

  function spawnCursor(x,y,speed){
    var count=speed>2?3:(speed>.5?2:1);
    for(var i=0;i<count;i++){
      if(cursorParts.length>=CURSOR_MAX)cursorParts.shift();
      cursorParts.push({x:x+rnd(-4,4),y:y+rnd(-4,4),vx:rnd(-.5,.5),vy:rnd(-.5,.5),size:.6+Math.random()*2,life:1,decay:.014+Math.random()*.028,pink:Math.random()<.6});

    if(mouseOn){
      trail.push({x:mouseX,y:mouseY});if(trail.length>MAX_TRAIL)trail.shift();

      if(trail.length>2){var step=Math.max(2,Math.floor(trail.length/5));for(var k=0;k<trail.length-step;k+=step){if(Math.random()<.3)spawnCursor(trail[k].x,trail[k].y,1)}}

    for(var j=cursorParts.length-1;j>=0;j--){var cp=cursorParts[j];cp.x+=cp.vx;cp.y+=cp.vy;cp.vx*=.965;cp.vy*=.965;cp.life-=cp.decay;if(cp.life<=0)cursorParts.splice(j,1)}
    var samples=[];if(mouseOn)samples.push({x:mouseX,y:mouseY});

  }
