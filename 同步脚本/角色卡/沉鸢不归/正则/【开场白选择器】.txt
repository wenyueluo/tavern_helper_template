/* ── 噪声函数（有机燃烧边缘）── */
function hash2D(x,y){ var n=Math.sin(x*127.1+y*311.7)*43758.5453; return n-Math.floor(n); }
function noise2D(x,y){
  var ix=Math.floor(x),iy=Math.floor(y),fx=x-ix,fy=y-iy;
  fx=fx*fx*(3-2*fx); fy=fy*fy*(3-2*fy);
  var a=hash2D(ix,iy),b=hash2D(ix+1,iy),c=hash2D(ix,iy+1),d=hash2D(ix+1,iy+1);
  return a+(b-a)*fx+(c-a)*fy+(a-b-c+d)*fx*fy;
}
function fbm2D(x,y,octaves){
  var v=0,amp=1,freq=1,tot=0;
  for(var i=0;i<octaves;i++){ v+=amp*noise2D(x*freq,y*freq); tot+=amp; amp*=0.5; freq*=2; }
  return v/tot;
}

/* ══════════════════════════════════════
   燃烧蔓延 — 基于噪声的有机边缘
   ══════════════════════════════════════ */
var burnThreshold;  /* 每格的燃烧到达阈值[0,1]：右下角≈0最先烧，左上角≈1最后烧 */

/* 预计算确定性燃烧阈值：以右下角为火源，距离+噪声 → 有机火线 */
function computeBurnThresholds(){
  burnThreshold=new Float32Array(GRID_W*GRID_H);
  var maxd=0, i;
  /* 第一遍：距离场（UV空间，火源在右下角(1,1)）+ 噪声扰动 */
  for(var row=0;row<GRID_H;row++){
    for(var col=0;col<GRID_W;col++){
      var u=col/(GRID_W-1), v=row/(GRID_H-1);
      var du=1-u, dv=1-v;
      var dist=Math.sqrt(du*du+dv*dv);                 /* 0(角)→~1.414(对角) */
      var n=fbm2D(col*.06, row*.06, 4);                /* 有机扰动 */
      var t=dist + (n-.5)*.42;                         /* 噪声让火线不规则 */
      if(t<0) t=0;
      burnThreshold[row*GRID_W+col]=t;
      if(t>maxd) maxd=t;
    }
  }
  /* 第二遍：归一化到[0,1]，保证最后一格阈值=1 → p=1时必定烧尽 */
  if(maxd<=0) maxd=1;
  for(i=0;i<burnThreshold.length;i++) burnThreshold[i]/=maxd;
}

function initBurnGrid(){
  burnGrid=new Float32Array(GRID_W*GRID_H);
  computeBurnThresholds();
  burnDone=false; burnActive=false; burnStarted=false; burnCooling=false;
  flames=[]; embers=[]; smokes=[]; ashes=[];
}

function igniteCell(col,row,val){
  if(col<0||col>=GRID_W||row<0||row>=GRID_H) return;
  var idx=row*GRID_W+col;
  if(burnGrid[idx]<=0) burnGrid[idx]=val||.012;
}

/* 确定性燃烧：全局进度 p∈[0,1] 在 BURN_DURATION 内匀速走完，
   每格燃烧度 = (火线位置 - 该格阈值)/前沿宽度，p=1 时全部=1 必定烧尽。
   时长由 BURN_DURATION 精确控制，不靠概率、不留死角、不需强停。 */
var BURN_FRONT=0.16;   /* 火焰前沿宽度（占总进度比例）：越小火线越锐利 */
function updateBurnGrid(dt){
  if(!burnActive||!burnThreshold) return;
  var elapsed=performance.now()-burnStartTime;
  /* 火线推进到 1+BURN_FRONT，确保阈值=1 的格子也能烧到满 */
  var p=Math.min(1, elapsed/BURN_DURATION)*(1+BURN_FRONT);

  for(var i=0;i<burnGrid.length;i++){
    var b=(p-burnThreshold[i])/BURN_FRONT;   /* 火线扫过该格的进度 */
    if(b<0) b=0; else if(b>1) b=1;
    if(b>burnGrid[i]) burnGrid[i]=b;          /* 只增不减，单调烧穿 */
  }
}

/* ══════════════════════════════════════
   粒子系统 — 火焰 / 余烬 / 烟雾
   ══════════════════════════════════════ */
function gridToScreen(col,row){
  return {
    x: (col+.5)/GRID_W*burnPaperCanvas.width,
    y: (row+.5)/GRID_H*burnPaperCanvas.height
  };
}

function spawnFlameAt(sx,sy,intensity){
  if(flames.length>=FLAME_MAX) flames.splice(0,20);
  var count=Math.floor(3+intensity*6);
  for(var i=0;i<count;i++){
    flames.push({
      x:sx+rnd(-10,10), y:sy+rnd(-4,2), px:sx, py:sy,
      vx:rnd(-1.2,1.2), vy:-(rnd(1.8,4.5)+intensity*rnd(1,3)),
      life:1, decay:rnd(.012,.028),
      size:rnd(6,16)+intensity*rnd(0,8),
      flick:rnd(0,6.28), sway:rnd(.3,1.3), intensity:intensity
    });
  }
}

function spawnEmberAt(sx,sy){
  if(embers.length>=EMBER_MAX) embers.splice(0,15);
  embers.push({
    x:sx+rnd(-8,8), y:sy, px:sx, py:sy,
    vx:rnd(-2.6,2.6), vy:-(rnd(4.5,11)),
    life:1, decay:rnd(.006,.018), size:rnd(.6,2.2),
    flick:rnd(0,6.28), sway:rnd(.8,2.8)
  });
}

function spawnSmokeAt(sx,sy,sizeScale){
  if(smokes.length>=SMOKE_MAX) smokes.splice(0,8);
  smokes.push({
    x:sx+rnd(-14,14), y:sy, px:sx, py:sy,
    vx:rnd(-.6,.6), vy:-(rnd(.7,1.6)),
    life:1, decay:rnd(.003,.007), size:rnd(10,24)*(sizeScale||1),
    flick:rnd(0,6.28), sway:rnd(.2,.8)
  });
}

function spawnAshAt(sx,sy){
  if(ashes.length>=ASH_MAX) return;
  ashes.push({
    x:sx, y:sy, px:sx, py:sy,
    vx:rnd(-.3,.3), vy:rnd(-.2,.5),
    life:1, decay:rnd(.0008,.003), size:rnd(1.5,5),
    alpha:.7+Math.random()*.3
  });
}

function updateParticles(dt){
  var dtSec=Math.min(dt/1000,.05), mult=dtSec*60;

  /* 火焰 */
  for(var i=flames.length-1;i>=0;i--){
    var f=flames[i];
    f.flick+=.22*mult;
    f.px=f.x; f.py=f.y;
    f.x+=f.vx*mult+Math.sin(f.flick)*f.sway*mult;
    f.y+=f.vy*mult;
    f.vy-=.04*mult; f.life-=f.decay*mult; f.size+=.04*mult;
    if(f.life<=0){ flames.splice(i,1); }
  }

  /* 余烬 */
  for(var j=embers.length-1;j>=0;j--){
    var e=embers[j];
    e.flick+=.18*mult;
    e.px=e.x; e.py=e.y;
    e.x+=e.vx*mult+Math.sin(e.flick)*e.sway*mult;
    e.y+=e.vy*mult;
    e.vy-=.015*mult; e.life-=e.decay*mult;
    if(e.life<=0){ embers.splice(j,1); }
  }

  /* 烟雾 */
  for(var k=smokes.length-1;k>=0;k--){
    var sm=smokes[k];
    sm.flick+=.015*mult;
    sm.px=sm.x; sm.py=sm.y;
    sm.x+=sm.vx*mult+Math.sin(sm.flick)*sm.sway*mult;
    sm.y+=sm.vy*mult;
    sm.vy*=.994; sm.size+=.3*mult; sm.life-=sm.decay*mult;
    if(sm.life<=0){ smokes.splice(k,1); }
  }

  /* 灰烬碎片 */
  for(var a=ashes.length-1;a>=0;a--){
    var ash=ashes[a];
    ash.px=ash.x; ash.py=ash.y;
    ash.x+=ash.vx*mult; ash.y+=ash.vy*mult;
    ash.life-=ash.decay*mult;
    if(ash.life<=0){ ashes.splice(a,1); }
  }
}

/* ══════════════════════════════════════
   渲染 — 燃烧遮罩（低分辨率→放大）
   ══════════════════════════════════════ */
/* 把燃烧网格生成一张低分辨率 RGBA 图层（fn(burn)→[r,g,b,a]），供平滑放大叠加 */
function buildGridLayer(fn){
  burnMaskCanvas.width=GRID_W; burnMaskCanvas.height=GRID_H;
  var img=burnMaskCtx.createImageData(GRID_W,GRID_H);
  var n=GRID_W*GRID_H;
  for(var i=0;i<n;i++){
    var burn=burnGrid?burnGrid[i]:0;
    var c=fn(burn);
    var pi=i*4;
    img.data[pi]=c[0]; img.data[pi+1]=c[1]; img.data[pi+2]=c[2]; img.data[pi+3]=c[3];
  }
  burnMaskCtx.putImageData(img,0,0);
}

/* 离屏：CSS mask 图（白=保留封面, 黑/透明=挖空露出目录） */
var maskUrlCanvas=document.createElement('canvas');
var maskUrlCtx=maskUrlCanvas.getContext('2d');

function renderBurnMask(){
  var W=burnPaperCanvas.width, H=burnPaperCanvas.height;
  if(W<10||H<10) return;
  var ctx=burnPaperCtx;

  /* ── A. 更新真封面 DOM 的 mask：烧尽处变透明 → 露出下层目录 ── */
  /* mask 用 alpha：完好=不透明(保留)，烧尽=透明(挖空) */
  buildGridLayer(function(burn){
    if(burn<.74) return [255,255,255,255];          /* 保留封面 */
    var t=Math.min(1,(burn-.74)/.24);
    return [255,255,255, Math.round(255*(1-t))];    /* 渐挖空 */
  });
  /* 平滑放大到适配 DOM 比例的 mask 图 */
  maskUrlCanvas.width=GRID_W; maskUrlCanvas.height=GRID_H;
  maskUrlCtx.clearRect(0,0,GRID_W,GRID_H);
  maskUrlCtx.drawImage(burnMaskCanvas,0,0);
  var url='url('+maskUrlCanvas.toDataURL()+')';
  coverLayer.css({'-webkit-mask-image':url,'mask-image':url});

  /* ── B. 在焦痕画布上只画"焦褐/炭黑"边缘（叠在真封面之上）── */
  ctx.globalCompositeOperation='source-over';
  ctx.clearRect(0,0,W,H);
  buildGridLayer(function(burn){
    if(burn<.06) return [0,0,0,0];                  /* 完好：透明，露出真封面字 */
    if(burn<.32){                                    /* 焦化：半透明褐 */
      var t=(burn-.06)/.26;
      return [130,75,22, Math.round(200*t)];
    }
    if(burn<.74){                                    /* 炭化：褐→黑 */
      var t2=(burn-.32)/.42;
      return [130*(1-t2)+18*t2, 75*(1-t2)+10*t2, 22*(1-t2)+6*t2, 235];
    }
    var t3=Math.min(1,(burn-.74)/.24);               /* 烧尽：炭黑渐隐(被挖空区) */
    return [18,10,6, Math.round(235*(1-t3))];
  });
  ctx.imageSmoothingEnabled=true;
  ctx.drawImage(burnMaskCanvas,0,0,GRID_W,GRID_H,0,0,W,H);
}

/* ── 燃烧边缘辉光叠加 ── */
function renderBurnEdgeGlow(){
  var W=burnPaperCanvas.width, H=burnPaperCanvas.height;
  if(W<10||H<10) return;
  burnPaperCtx.globalCompositeOperation='lighter';
  for(var row=0;row<GRID_H;row+=2){
    for(var col=0;col<GRID_W;col+=2){
      var burn=burnGrid[row*GRID_W+col];
      if(burn>.04&&burn<.24){
        var sx=(col+.5)/GRID_W*W, sy=(row+.5)/GRID_H*H;
        var glowAlpha=(.24-burn)/.2*.35;
        var glow=burnPaperCtx.createRadialGradient(sx,sy,0,sx,sy,12);
        glow.addColorStop(0,'rgba(255,160,30,'+glowAlpha+')');
        glow.addColorStop(1,'transparent');
        burnPaperCtx.fillStyle=glow;
        burnPaperCtx.fillRect(sx-14,sy-14,28,28);
      }
    }
  }
  burnPaperCtx.globalCompositeOperation='source-over';
}

/* ══════════════════════════════════════
   渲染 — 火焰粒子画布
   ══════════════════════════════════════ */
function renderFireParticles(){
  fireCtx.globalCompositeOperation='source-over';
  fireCtx.clearRect(0,0,fireCanvas.width,fireCanvas.height);

  var W=fireCanvas.width, H=fireCanvas.height;

  /* 烟雾（正常混合） */
  for(var i=0;i<smokes.length;i++){
    var sm=smokes[i];
    var a=sm.life*.12;
    if(a<=0) continue;
    var g=fireCtx.createRadialGradient(sm.x,sm.y,0,sm.x,sm.y,sm.size);
    var sh=Math.round(22+sm.life*16);
    g.addColorStop(0,'rgba('+sh+','+(sh-4)+','+(sh+6)+','+a+')');
    g.addColorStop(1,'rgba('+sh+','+(sh-4)+','+(sh+6)+',0)');
    fireCtx.fillStyle=g; fireCtx.beginPath(); fireCtx.arc(sm.x,sm.y,sm.size,0,6.2832); fireCtx.fill();
  }

  /* 火焰+余烬（叠加发光） */
  fireCtx.globalCompositeOperation='lighter';

  /* 火焰粒子 */
  for(var j=0;j<flames.length;j++){
    var f=flames[j];
    var L=f.life;
    if(L<=.01) continue;
    /* 温度→颜色：白热核心→橙中层→暗红尖端 */
    var hue=10+L*42; var sat=95+(1-L)*5; var light=30+L*55;
    var a=L*.58;
    var g=fireCtx.createRadialGradient(f.x,f.y,0,f.x,f.y,f.size);
    g.addColorStop(0,'hsla('+hue+','+sat+'%,'+Math.min(98,light+12)+'%,'+a+')');
    g.addColorStop(.35,'hsla('+hue+','+sat+'%,'+light+'%,'+(a*.7)+')');
    g.addColorStop(1,'hsla('+(hue-8)+','+sat+'%,'+(light*.45)+'%,0)');
    fireCtx.fillStyle=g; fireCtx.beginPath(); fireCtx.arc(f.x,f.y,f.size,0,6.2832); fireCtx.fill();
  }

  /* 余烬粒子 */
  for(var k=0;k<embers.length;k++){
    var e=embers[k];
    var eL=e.life;
    if(eL<=.01) continue;
    /* 拖尾 */
    var trailAlpha=eL*.45;
    fireCtx.strokeStyle='hsla(30,90%,'+Math.round(50+eL*40)+'%,'+trailAlpha+')';
    fireCtx.lineWidth=e.size*.7;
    fireCtx.beginPath(); fireCtx.moveTo(e.px,e.py); fireCtx.lineTo(e.x,e.y); fireCtx.stroke();
    /* 发光点 */
    fireCtx.fillStyle='hsla(32,85%,'+Math.round(60+eL*32)+'%,'+eL*.75+')';
    fireCtx.beginPath(); fireCtx.arc(e.x,e.y,e.size,0,6.2832); fireCtx.fill();
  }

  /* 底部热浪辉光 */
  var now=performance.now();
  var flicker=.85+.15*Math.sin(now*.012)*Math.sin(now*.019);
  var glowGrad=fireCtx.createLinearGradient(0,H*.5,0,H);
  glowGrad.addColorStop(0,'rgba(255,120,15,0)');
  glowGrad.addColorStop(.5,'rgba(255,90,10,'+(.08*flicker)+')');
  glowGrad.addColorStop(1,'rgba(255,50,5,'+(.18*flicker)+')');
  fireCtx.fillStyle=glowGrad; fireCtx.fillRect(0,0,W,H);

  fireCtx.globalCompositeOperation='source-over';
}

/* ══════════════════════════════════════
   灰烬碎片渲染（纸张燃尽后残留）
   ══════════════════════════════════════ */
function renderAshes(ctx,W,H){
  for(var i=0;i<ashes.length;i++){
    var ash=ashes[i];
    var a=ash.life*ash.alpha;
    if(a<=.01) continue;
    /* 不规则灰烬片 */
    ctx.save(); ctx.translate(ash.x,ash.y); ctx.rotate(ash.life*2.5);
    ctx.fillStyle='rgba(35,25,15,'+a+')';
    ctx.beginPath();
    var s=ash.size;
    ctx.moveTo(-s,-s*.4); ctx.lineTo(s*.3,-s*.7); ctx.lineTo(s*.8,-s*.2);
    ctx.lineTo(s*.6,s*.5); ctx.lineTo(-s*.2,s*.7); ctx.lineTo(-s*.7,s*.1);
    ctx.closePath(); ctx.fill();
    /* 微弱的余烬光 */
    if(a>.3){
      ctx.fillStyle='rgba(255,120,30,'+(a*.15)+')';
      ctx.beginPath(); ctx.arc(0,0,s*.3,0,6.2832); ctx.fill();
    }
    ctx.restore();
  }
}

/* ══════════════════════════════════════
   主循环
   ══════════════════════════════════════ */
var burnLoopId=null;

function burnLoop(timestamp){
  if(burnDone){ burnLoopId=null; return; }
  var dt=16; /* 近似 60fps */

  if(burnActive){
    updateBurnGrid(dt);
    updateParticles(dt);

    /* 沿燃烧边缘生成火焰粒子 */
    var now=performance.now();
    var elapsed=now-burnStartTime;
    var partCount=0;
    for(var row=0;row<GRID_H;row+=2){
      for(var col=0;col<GRID_W;col+=2){
        var burn=burnGrid[row*GRID_W+col];
        if(burn>.03&&burn<.35){
          var pos=gridToScreen(col,row);
          var intensity=.15+(.35-burn)/.32*.85;
          spawnFlameAt(pos.x,pos.y,intensity);
          partCount++;
          if(partCount>80) break;
        }
      }
      if(partCount>80) break;
    }

    /* 燃烧边缘脱落余烬 */
    for(var er=0;er<GRID_H;er+=3){
      for(var ec=0;ec<GRID_W;ec+=3){
        var eb=burnGrid[er*GRID_W+ec];
        if(eb>.08&&eb<.4&&Math.random()<.25){
          var epos=gridToScreen(ec,er);
          spawnEmberAt(epos.x,epos.y);
        }
      }
    }

    /* 烟雾从燃烧区域升起 */
    if(Math.random()<.5){
      for(var sr=0;sr<GRID_H;sr+=4){
        for(var sc=0;sc<GRID_W;sc+=4){
          var sb=burnGrid[sr*GRID_W+sc];
          if(sb>.15&&sb<.7&&Math.random()<.3){
            var spos=gridToScreen(sc,sr);
            spawnSmokeAt(spos.x,spos.y,1+sb*.5);
          }
        }
      }
    }

    /* 燃尽单元格 → 生成灰烬碎片 */
    for(var ar=0;ar<GRID_H;ar+=4){
      for(var ac=0;ac<GRID_W;ac+=4){
        var ab=burnGrid[ar*GRID_W+ac];
        if(ab>.75&&ab<.85&&Math.random()<.08){
          var apos=gridToScreen(ac,ar);
          spawnAshAt(apos.x,apos.y);
        }
      }
    }

    /* 渲染燃烧遮罩 */
    renderBurnMask();
    renderBurnEdgeGlow();

    /* 渲染火焰粒子 */
    renderFireParticles();

    /* 灰烬碎片渲染到纸张画布 */
    renderAshes(burnPaperCtx,burnPaperCanvas.width,burnPaperCanvas.height);

    /* 检查是否全部烧尽 → 进入冷却阶段（不强停、不抹封面） */
    var allBurned=true;
    for(var ci=0;ci<burnGrid.length;ci++){
      if(burnGrid[ci]<.99){ allBurned=false; break; }
    }
    if(allBurned){ enterCooling(); }
  }
  else if(burnCooling){
    /* 冷却：停止蔓延与生成新火焰，让残余火焰/余烬/灰烬自然飘散消失 */
    updateParticles(dt);
    /* 网格已全烧尽 → renderBurnMask 会把封面整张挖空，目录完整露出 */
    renderBurnMask();
    renderFireParticles();
    renderAshes(burnPaperCtx,burnPaperCanvas.width,burnPaperCanvas.height);
    /* 灰烬缓慢变暗 */
    for(var ai=0;ai<ashes.length;ai++) ashes[ai].alpha-=.0012;
    /* 火焰、余烬、烟雾都散尽 → 收场 */
    if(flames.length===0&&embers.length===0&&smokes.length===0){
      endBurn();
      return;
    }
  }

  burnLoopId=requestAnimationFrame(burnLoop);
}

/* 蔓延 → 冷却 的唯一入口：停止蔓延，但保留已烧尽的网格（封面已被挖空） */
function enterCooling(){
  if(burnCooling) return;
  burnActive=false; burnCooling=true;
}

/* ══════════════════════════════════════
   启动 / 停止
   ══════════════════════════════════════ */
function startBurn(){
  if(burnActive||burnStarted) return;   /* 已在燃烧流程中（含点火前等待）→ 不重入 */

  /* 开启新一轮运行：作废所有旧时间线 */
  burnRun++;
  var myRun=burnRun;
  clearBurnTimers();

  initBurnGrid();
  resizeBurnPaper();

  /* 显示焦痕画布；真封面 DOM 保留原位，用 mask 逐步挖空；目录在下层 */
  burnPaperCanvas.style.display='block';
  coverLayer.css({
    '-webkit-mask-size':'100% 100%','mask-size':'100% 100%',
    '-webkit-mask-repeat':'no-repeat','mask-repeat':'no-repeat',
    '-webkit-mask-image':'none','mask-image':'none'
  });
  coverLayer.addClass('burning');
  tocLayer.addClass('shown');

  /* 2秒后开始燃烧：火线从右下角匀速扫过，确定性地在 BURN_DURATION 内烧尽 */
  burnStarted=true;
  scheduleBurn(function(){
    burnActive=true; burnStartTime=performance.now();
    fireCanvas.style.display='block'; fireCanvas.style.opacity='1';
    /* 不再需要催熟/强停：updateBurnGrid 按时间进度保证 5s 内全部烧尽，
       烧尽后 burnLoop 自动进入冷却→收场（见 enterCooling/endBurn） */
  },BURN_DELAY,myRun);


  if(!burnLoopId) burnLoopId=requestAnimationFrame(burnLoop);
}

function endBurn(){
  if(burnDone) return;
  burnDone=true; burnActive=false; burnStarted=false; burnCooling=false;
  if(burnLoopId){ cancelAnimationFrame(burnLoopId); burnLoopId=null; }

  var myRun=burnRun;

  /* 火焰画布渐隐 */
  fireCanvas.style.opacity='0';
  scheduleBurn(function(){
    fireCanvas.style.display='none';
    try{fireCtx.clearRect(0,0,fireCanvas.width,fireCanvas.height);}catch(e){}
    fireCanvas.style.opacity='1';
  },600,myRun);

  /* 封面已整张烧尽（画布全透明，目录完整露出）→ 隐藏燃烧画布交还给 DOM 目录层 */
  scheduleBurn(function(){
    burnPaperCanvas.style.display='none';
    try{burnPaperCtx.clearRect(0,0,burnPaperCanvas.width,burnPaperCanvas.height);}catch(e){}
  },800,myRun);
}

/* ── 重置（返回封面）── */
function resetBurn(){
  /* 作废所有挂起时间线：代号自增 + 清 timer，旧 setTimeout 即使漏网也不执行 */
  burnRun++;
  clearBurnTimers();
  burnDone=true; burnActive=false; burnStarted=false; burnCooling=false;
  if(burnLoopId){ cancelAnimationFrame(burnLoopId); burnLoopId=null; }
  burnGrid=null; flames=[]; embers=[]; smokes=[]; ashes=[];
  fireCanvas.style.display='none';
  burnPaperCanvas.style.display='none';
  try{fireCtx.clearRect(0,0,fireCanvas.width,fireCanvas.height);}catch(e){}
  try{burnPaperCtx.clearRect(0,0,burnPaperCanvas.width,burnPaperCanvas.height);}catch(e){}
  coverLayer.removeClass('burning');
  coverLayer.css({'-webkit-mask-image':'none','mask-image':'none'});  /* 恢复完整封面 */
  tocLayer.removeClass('shown');
  burnDone=false;
}

coverBtn.on('click',function(e){ e.stopPropagation(); startBurn(); });

/* ═══════════════════════════════════════════
   光标拖尾粒子 + 环境飘落粒子（排斥避让）
   ═══════════════════════════════════════════ */
(function(){
  var container = document.querySelector(".kcb-wrap");
  var canvas = document.getElementById("kcb-particles-canvas");
  if(!container||!canvas) return;
  var ctx = canvas.getContext("2d");
  var w, h, animId;
  var cursorParts = [];
  var ambientParts = [];
  var trail = [];
  var MAX_TRAIL = 36;
  var AMBIENT_MAX = 55;
  var CURSOR_MAX = 80;
  var REPEL_R = 68;
  var REPEL_FORCE = 0.6;
  var mouseX = -200, mouseY = -200;
  var mouseOn = false;
  var mouseSpeed = 0;
  var lastMX = -200, lastMY = -200;

  function resize(){
    var r = container.getBoundingClientRect();
    w = canvas.width = r.width;
    h = canvas.height = r.height;
  }

  function rnd(a,b){ return a + Math.random()*(b-a); }

  function initAmbient(){
    ambientParts = [];
    for(var i=0; i<AMBIENT_MAX; i++){
      ambientParts.push({
        x: Math.random()*w,
        y: Math.random()*h,
        vx: (Math.random()-.5)*.25,
        vy: .15 + Math.random()*.5,
        _baseVy: .15 + Math.random()*.5,
        size: .5 + Math.random()*1.3,
        gold: Math.random() < .65,
        phase: Math.random()*Math.PI*2
      });
    }
  }

  function spawnCursor(x, y, speed){
    var count = speed > 2 ? 3 : (speed > .5 ? 2 : 1);
    for(var i=0; i<count; i++){
      if(cursorParts.length >= CURSOR_MAX) cursorParts.shift();
      var gold = Math.random() < .55;
      cursorParts.push({
        x: x + rnd(-5,5),
        y: y + rnd(-5,5),
        vx: rnd(-.6,.6),
        vy: rnd(-.6,.6),
        size: .7 + Math.random()*2.4,
        life: 1,
        decay: .012 + Math.random()*.032,
        gold: gold
      });
    }
  }

  function update(){
    if(mouseOn){
      trail.push({x: mouseX, y: mouseY});
      if(trail.length > MAX_TRAIL) trail.shift();
      var spd = mouseSpeed;
      var count = spd > 3 ? 4 : (spd > 1 ? 3 : 2);
      for(var i=0; i<count; i++) spawnCursor(mouseX, mouseY, spd);
      if(trail.length > 2){
        var step = Math.max(2, Math.floor(trail.length/6));
        for(var k=0; k<trail.length-step; k+=step){
          var tp = trail[k];
          if(Math.random() < .35) spawnCursor(tp.x, tp.y, 1);
        }
      }
    } else {
      if(trail.length > 0) trail.shift();
    }

    for(var j=cursorParts.length-1; j>=0; j--){
      var cp = cursorParts[j];
      cp.x += cp.vx;
      cp.y += cp.vy;
      cp.vx *= .965;
      cp.vy *= .965;
      cp.life -= cp.decay;
      if(cp.life <= 0) cursorParts.splice(j,1);
    }

    var samples = [];
    if(mouseOn) samples.push({x: mouseX, y: mouseY});
    for(var s=0; s<trail.length; s+=4) samples.push(trail[s]);

    for(var m=0; m<ambientParts.length; m++){
      var ap = ambientParts[m];
      if(samples.length > 0 && mouseOn){
        var totalFx = 0, totalFy = 0;
        for(var n=0; n<samples.length; n++){
          var dx = ap.x - samples[n].x;
          var dy = ap.y - samples[n].y;
          var dist = Math.sqrt(dx*dx + dy*dy);
          if(dist < REPEL_R && dist > .5){
            var f = REPEL_FORCE * Math.pow(1 - dist/REPEL_R, 1.5);
            totalFx += (dx/dist) * f;
            totalFy += (dy/dist) * f;
          }
        }
        ap.vx += totalFx;
        ap.vy += totalFy;
      }
      ap.x += ap.vx + Math.sin(ap.phase + performance.now()*.0008) * .12;
      ap.y += ap.vy;
      ap.phase += .008;
      ap.vx *= .985;
      ap.vy *= .985;
      ap.vy += (ap._baseVy - ap.vy) * .003;
      if(ap.y > h + 15){ ap.y = -15; ap.x = Math.random()*w; ap.vx = 0; ap.vy = ap._baseVy; }
      if(ap.x < -15) ap.x = w + 15;
      if(ap.x > w + 15) ap.x = -15;
    }

    if(mouseOn){
      mouseSpeed = Math.sqrt((mouseX-lastMX)*(mouseX-lastMX) + (mouseY-lastMY)*(mouseY-lastMY));
      lastMX = mouseX;
      lastMY = mouseY;
    }
  }

  function draw(){
    ctx.clearRect(0, 0, w, h);
    for(var i=0; i<ambientParts.length; i++){
      var ap = ambientParts[i];
      var a = .25 + Math.sin(ap.phase)*.12;
      var col = ap.gold ? "200,164,92" : "154,126,200";
      ctx.fillStyle = "rgba("+col+","+a+")";
      ctx.beginPath();
      ctx.arc(ap.x, ap.y, ap.size, 0, Math.PI*2);
      ctx.fill();
    }
    ctx.globalCompositeOperation = "lighter";
    for(var j=0; j<cursorParts.length; j++){
      var cp = cursorParts[j];
      var a2 = cp.life * .6;
      var col2 = cp.gold ? "200,164,92" : "154,126,200";
      var sz = cp.size * (1 + cp.life*.5);
      var g = ctx.createRadialGradient(cp.x, cp.y, 0, cp.x, cp.y, sz*2.5);
      g.addColorStop(0, "rgba("+col2+","+a2+")");
      g.addColorStop(.35, "rgba("+col2+","+(a2*.45)+")");
      g.addColorStop(1, "rgba("+col2+",0)");
      ctx.fillStyle = g;
      ctx.beginPath();
      ctx.arc(cp.x, cp.y, sz*2.5, 0, Math.PI*2);
      ctx.fill();
    }
    ctx.globalCompositeOperation = "source-over";
  }

  function loop(){
    update();
    draw();
    animId = requestAnimationFrame(loop);
  }

  container.addEventListener("mousemove", function(e){
    var r = canvas.getBoundingClientRect();
    mouseX = e.clientX - r.left;
    mouseY = e.clientY - r.top;
    mouseOn = true;
  });
  container.addEventListener("mouseenter", function(e){
    mouseOn = true;
    var r = canvas.getBoundingClientRect();
    mouseX = e.clientX - r.left;
    mouseY = e.clientY - r.top;
  });
  container.addEventListener("mouseleave", function(){ mouseOn = false; });
  window.addEventListener("resize", function(){ resize(); initAmbient(); });

  resize();
  initAmbient();
  loop();
})();

});
</script>
