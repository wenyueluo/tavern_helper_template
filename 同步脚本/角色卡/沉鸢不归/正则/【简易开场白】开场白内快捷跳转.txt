/* 读取当前 swipe_id（用于高亮） */
function currentSwipe(){
  try{
    if(typeof getCurrentMessageId==='function' && typeof getChatMessages==='function'){
      var id=getCurrentMessageId();
      var m=getChatMessages(id,{include_swipes:true});
      if(m&&m[0]&&typeof m[0].swipe_id==='number') return m[0].swipe_id;
    }
  }catch(e){}
  return -1;
}

/* 跳转到指定开场白（await + 强制刷新） */
var busy=false;
function goto(swipe){
  if(busy)return; busy=true;
  function done(){busy=false;}
  try{
    if(typeof setChatMessages==='function'){
      Promise.resolve(setChatMessages([{message_id:0,swipe_id:swipe}],{refresh:'all'})).then(done,done);
    }else if(typeof SillyTavern!=='undefined'&&SillyTavern.getContext){
      var c=SillyTavern.getContext();
      if(c&&c.setSwipe){c.setSwipe(swipe);setTimeout(done,800);}else done();
    }else if(window.TavernHelper&&window.TavernHelper.switchMessage){
      window.TavernHelper.switchMessage(swipe);done();
    }else done();
  }catch(err){done();}
}

var wrap=$('#jk-wrap');
var cur=currentSwipe();

function mkBtn(o,isSpecial){
  var cls='jk-num'+(isSpecial?' special':'')+(o.n===cur?' active':'');
  return $('<div class="'+cls+'" data-swipe="'+o.n+'">'+o.n
    +'<span class="tip"><b>'+o.name+'</b><i>'+o.sub+'</i></span></div>');
}

wrap.append('<span class="jk-div"></span>');
for(var j=0;j<specials.length;j++) wrap.append(mkBtn(specials[j],true));

wrap.on('click','.jk-num',function(){ goto(parseInt($(this).data('swipe'),10)); });
$('#jk-back').on('click',function(){ goto(0); });

});
