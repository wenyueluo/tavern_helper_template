  // 将触发短语追加到酒馆输入框（iframe → 父窗口）
  function appendToInput(phrase){
    var sel='#send_textarea';
    var doc=null;
    try{ doc=window.parent&&window.parent.document; }catch(e){}
    var ta=null;
    if(doc){ ta=doc.querySelector(sel); }
    if(!ta){ try{ ta=document.querySelector(sel); }catch(e){} }
    if(!ta) return false;
    var v=ta.value||'';
    if(v && !/\s$/.test(v)) v+=' ';
    ta.value=v+phrase;
    try{ ta.dispatchEvent(new Event('input',{bubbles:true})); }catch(e){}
    try{ ta.focus(); }catch(e){}
    return true;
  }

  function bindRecall(){

          if(ok){ b.classList.remove('flash'); void b.offsetWidth; b.classList.add('flash'); }
        };
      })(btns[i]);
    }
  }

/* ═══════════════════════════════════════════
   光标拖尾粒子 + 环境飘落粒子（排斥避让）
   ═══════════════════════════════════════════ */
(function(){
  var container = document.getElementById("sb-wrap");
  var canvas = document.getElementById("sb-particles-canvas");
  if(!container||!canvas) return;
  var ctx = canvas.getContext("2d");
  var w, h, animId;
  var cursorParts = [];
  var ambientParts = [];
  var trail = [];
  var MAX_TRAIL = 36;
  var AMBIENT_MAX = 55;
  var CURSOR_MAX = 80;
  var REPEL_R = 68;
  var REPEL_FORCE = 0.6;
  var mouseX = -200, mouseY = -200;
  var mouseOn = false;
  var mouseSpeed = 0;
  var lastMX = -200, lastMY = -200;

  function resize(){
    var r = container.getBoundingClientRect();
    w = canvas.width = r.width;
    h = canvas.height = r.height;
  }

  function rnd(a,b){ return a + Math.random()*(b-a); }

  function initAmbient(){
    ambientParts = [];
    for(var i=0; i<AMBIENT_MAX; i++){
      ambientParts.push({
        x: Math.random()*w,
        y: Math.random()*h,
        vx: (Math.random()-.5)*.25,
        vy: .15 + Math.random()*.5,
        _baseVy: .15 + Math.random()*.5,
        size: .5 + Math.random()*1.3,
        gold: Math.random() < .65,
        phase: Math.random()*Math.PI*2
      });
    }
  }

  function spawnCursor(x, y, speed){
    var count = speed > 2 ? 3 : (speed > .5 ? 2 : 1);
    for(var i=0; i<count; i++){
      if(cursorParts.length >= CURSOR_MAX) cursorParts.shift();
      var gold = Math.random() < .55;
      cursorParts.push({
        x: x + rnd(-5,5),
        y: y + rnd(-5,5),
        vx: rnd(-.6,.6),
        vy: rnd(-.6,.6),
        size: .7 + Math.random()*2.4,
        life: 1,
        decay: .012 + Math.random()*.032,
        gold: gold
      });
    }
  }

  function update(){
    if(mouseOn){
      trail.push({x: mouseX, y: mouseY});
      if(trail.length > MAX_TRAIL) trail.shift();
      var spd = mouseSpeed;
      var count = spd > 3 ? 4 : (spd > 1 ? 3 : 2);
      for(var i=0; i<count; i++) spawnCursor(mouseX, mouseY, spd);
      if(trail.length > 2){
        var step = Math.max(2, Math.floor(trail.length/6));
        for(var k=0; k<trail.length-step; k+=step){
          var tp = trail[k];
          if(Math.random() < .35) spawnCursor(tp.x, tp.y, 1);
        }
      }
    } else {
      if(trail.length > 0) trail.shift();
    }

    for(var j=cursorParts.length-1; j>=0; j--){
      var cp = cursorParts[j];
      cp.x += cp.vx;
      cp.y += cp.vy;
      cp.vx *= .965;
      cp.vy *= .965;
      cp.life -= cp.decay;
      if(cp.life <= 0) cursorParts.splice(j,1);
    }

    var samples = [];
    if(mouseOn) samples.push({x: mouseX, y: mouseY});
    for(var s=0; s<trail.length; s+=4) samples.push(trail[s]);

    for(var m=0; m<ambientParts.length; m++){
      var ap = ambientParts[m];
      if(samples.length > 0 && mouseOn){
        var totalFx = 0, totalFy = 0;
        for(var n=0; n<samples.length; n++){
          var dx = ap.x - samples[n].x;
          var dy = ap.y - samples[n].y;
          var dist = Math.sqrt(dx*dx + dy*dy);
          if(dist < REPEL_R && dist > .5){
            var f = REPEL_FORCE * Math.pow(1 - dist/REPEL_R, 1.5);
            totalFx += (dx/dist) * f;
            totalFy += (dy/dist) * f;
          }
        }
        ap.vx += totalFx;
        ap.vy += totalFy;
      }
      ap.x += ap.vx + Math.sin(ap.phase + performance.now()*.0008) * .12;
      ap.y += ap.vy;
      ap.phase += .008;
      ap.vx *= .985;
      ap.vy *= .985;
      ap.vy += (ap._baseVy - ap.vy) * .003;
      if(ap.y > h + 15){ ap.y = -15; ap.x = Math.random()*w; ap.vx = 0; ap.vy = ap._baseVy; }
      if(ap.x < -15) ap.x = w + 15;
      if(ap.x > w + 15) ap.x = -15;
    }

    if(mouseOn){
      mouseSpeed = Math.sqrt((mouseX-lastMX)*(mouseX-lastMX) + (mouseY-lastMY)*(mouseY-lastMY));
      lastMX = mouseX;
      lastMY = mouseY;
    }
  }

  function draw(){
    ctx.clearRect(0, 0, w, h);
    for(var i=0; i<ambientParts.length; i++){
      var ap = ambientParts[i];
      var a = .25 + Math.sin(ap.phase)*.12;
      var col = ap.gold ? "200,164,92" : "154,126,200";
      ctx.fillStyle = "rgba("+col+","+a+")";
      ctx.beginPath();
      ctx.arc(ap.x, ap.y, ap.size, 0, Math.PI*2);
      ctx.fill();
    }
    ctx.globalCompositeOperation = "lighter";
    for(var j=0; j<cursorParts.length; j++){
      var cp = cursorParts[j];
      var a2 = cp.life * .6;
      var col2 = cp.gold ? "200,164,92" : "154,126,200";
      var sz = cp.size * (1 + cp.life*.5);
      var g = ctx.createRadialGradient(cp.x, cp.y, 0, cp.x, cp.y, sz*2.5);
      g.addColorStop(0, "rgba("+col2+","+a2+")");
      g.addColorStop(.35, "rgba("+col2+","+(a2*.45)+")");
      g.addColorStop(1, "rgba("+col2+",0)");
      ctx.fillStyle = g;
      ctx.beginPath();
      ctx.arc(cp.x, cp.y, sz*2.5, 0, Math.PI*2);
      ctx.fill();
    }
    ctx.globalCompositeOperation = "source-over";
  }

  function loop(){
    update();
    draw();
    animId = requestAnimationFrame(loop);
  }

  container.addEventListener("mousemove", function(e){
    var r = canvas.getBoundingClientRect();
    mouseX = e.clientX - r.left;
    mouseY = e.clientY - r.top;
    mouseOn = true;
  });
  container.addEventListener("mouseenter", function(e){
    mouseOn = true;
    var r = canvas.getBoundingClientRect();
    mouseX = e.clientX - r.left;
    mouseY = e.clientY - r.top;
  });
  container.addEventListener("mouseleave", function(){ mouseOn = false; });
  window.addEvenatListener("resize", function(){ resize(); initAmbient(); });

  resize();
  initAmbient();
  loop();
})();

})();
</script>
</body>
</html>
